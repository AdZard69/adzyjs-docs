### **1. Interactive Character Showcase**

Create a dynamic showcase of different Minecraft characters with custom skins.

```jsx
import { Player } from 'adzyjs';
import { useState } from 'react';

function CharacterShowcase() {
  const characters = [
    {
      name: "Steve",
      skin: "https://example.com/steve.png",
      description: "The classic Minecraft character"
    },
    {
      name: "Alex", 
      skin: "https://example.com/alex.png",
      description: "The adventurous explorer"
    },
    {
      name: "Custom Hero",
      skin: "https://example.com/hero.png", 
      description: "Your custom creation"
    }
  ];

  const [selectedCharacter, setSelectedCharacter] = useState(0);

  return (
    <div className="character-showcase">
      <div className="character-display">
        <Player 
          skinUrl={characters[selectedCharacter].skin}
          animation="idle"
          width={400}
          height={600}
        />
      </div>
      
      <div className="character-selector">
        {characters.map((char, index) => (
          <button
            key={index}
            onClick={() => setSelectedCharacter(index)}
            className={`character-btn ${selectedCharacter === index ? 'active' : ''}`}
          >
            <img src={char.skin} alt={char.name} />
            <h3>{char.name}</h3>
            <p>{char.description}</p>
          </button>
        ))}
      </div>
    </div>
  );
}
```

### **2. Animated Landing Page**

Create an impressive landing page with animated Minecraft characters.

```jsx
import { Player } from 'adzyjs';
import { useEffect, useState } from 'react';

function AnimatedLanding() {
  const [currentAnimation, setCurrentAnimation] = useState('walking');
  
  useEffect(() => {
    const animations = ['walking', 'running', 'jumping', 'idle'];
    const interval = setInterval(() => {
      setCurrentAnimation(animations[Math.floor(Math.random() * animations.length)]);
    }, 3000);
    
    return () => clearInterval(interval);
  }, []);

  return (
    <div className="landing-page">
      <div className="hero-section">
        <div className="content">
          <h1>Welcome to My Portfolio</h1>
          <p>Interactive 3D experiences with AdzyJS</p>
          <button className="cta-button">Explore Projects</button>
        </div>
        
        <div className="character-hero">
          <Player 
            skinUrl="https://example.com/your-skin.png"
            animation={currentAnimation}
            width={300}
            height={400}
          />
        </div>
      </div>
    </div>
  );
}
```

### **Contact Form with Character**

```jsx
import { Player } from 'adzyjs';
import { useState } from 'react';

function ContactSection() {
  const [formData, setFormData] = useState({
    name: '',
    email: '',
    message: ''
  });
  const [isSubmitting, setIsSubmitting] = useState(false);

  const handleSubmit = async (e) => {
    e.preventDefault();
    setIsSubmitting(true);
    
    // Simulate form submission
    await new Promise(resolve => setTimeout(resolve, 2000));
    
    setIsSubmitting(false);
    alert('Message sent! I\'ll get back to you soon.');
    setFormData({ name: '', email: '', message: '' });
  };

  return (
    <div className="contact-section">
      <div className="contact-character">
        <Player 
          skinUrl="https://example.com/contact-character.png"
          animation={isSubmitting ? "running" : "idle"}
          width={300}
          height={400}
        />
        <div className="speech-bubble">
          <p>{isSubmitting ? "Sending your message..." : "Let's work together!"}</p>
        </div>
      </div>
      
      <form onSubmit={handleSubmit} className="contact-form">
        <h2>Get In Touch</h2>
        
        <input
          type="text"
          placeholder="Your Name"
          value={formData.name}
          onChange={(e) => setFormData({...formData, name: e.target.value})}
          required
        />
        
        <input
          type="email"
          placeholder="Your Email"
          value={formData.email}
          onChange={(e) => setFormData({...formData, email: e.target.value})}
          required
        />
        
        <textarea
          placeholder="Your Message"
          value={formData.message}
          onChange={(e) => setFormData({...formData, message: e.target.value})}
          rows={5}
          required
        />
        
        <button type="submit" disabled={isSubmitting}>
          {isSubmitting ? "Sending..." : "Send Message"}
        </button>
      </form>
    </div>
  );
}
```

## ðŸ”§ Performance Optimization

### **Lazy Loading Characters**

```jsx
import { lazy, Suspense } from 'react';

const Player = lazy(() => import('adzyjs').then(module => ({ default: module.Player })));

function OptimizedCharacter({ skinUrl, ...props }) {
  return (
    <Suspense fallback={<div className="character-loading">Loading...</div>}>
      <Player skinUrl={skinUrl} {...props} />
    </Suspense>
  );
}
```

### **Image Optimization**

```jsx
function OptimizedSkinLoader({ skinUrl, fallback }) {
  const [imageSrc, setImageSrc] = useState(fallback);
  const [isLoading, setIsLoading] = useState(true);

  useEffect(() => {
    const img = new Image();
    img.onload = () => {
      setImageSrc(skinUrl);
      setIsLoading(false);
    };
    img.onerror = () => {
      setIsLoading(false);
    };
    img.src = skinUrl;
  }, [skinUrl]);

  return (
    <div className={`character-container ${isLoading ? 'loading' : ''}`}>
      <Player skinUrl={imageSrc} />
    </div>
  );
}
```
