## Dynamic Theme Switching

```jsx
import { useState } from 'react';

function App() {
  const [theme, setTheme] = useState('default');

  const themes = [
    { name: 'Default', value: 'default' },
    { name: 'Dark', value: 'dark-theme' },
    { name: 'Professional', value: 'professional-theme' },
    { name: 'Gaming', value: 'gaming-theme' }
  ];

  return (
    <div className={`portfolio ${theme}`}>
      {/* Theme Selector */}
      <div className="theme-selector">
        {themes.map(t => (
          <button
            key={t.value}
            onClick={() => setTheme(t.value)}
            className={theme === t.value ? 'active' : ''}
          >
            {t.name}
          </button>
        ))}
      </div>

      {/* Your portfolio content */}
    </div>
  );
}
```

## Custom Character Animations

### Animation State Management

```jsx
import { useState, useEffect } from 'react';
import { Player } from 'adzyjs';

function AnimatedHero() {
  const [animation, setAnimation] = useState('idle');
  const [isHovered, setIsHovered] = useState(false);

  const animations = ['idle', 'wave', 'jump', 'dance', 'thumbsup'];

  useEffect(() => {
    if (!isHovered) {
      const interval = setInterval(() => {
        setAnimation(animations[Math.floor(Math.random() * animations.length)]);
      }, 4000);
      return () => clearInterval(interval);
    }
  }, [isHovered]);

  return (
    <div 
      className="hero-character"
      onMouseEnter={() => {
        setIsHovered(true);
        setAnimation('wave');
      }}
      onMouseLeave={() => {
        setIsHovered(false);
        setAnimation('idle');
      }}
    >
      <Player 
        skinUrl="https://example.com/your-skin.png"
        animation={animation}
        width={250}
        height={350}
      />
    </div>
  );
}
```

### Scroll-Based Animations

```jsx
import { useEffect, useState } from 'react';

function ScrollAnimatedCharacter({ skinUrl, baseAnimation = 'idle' }) {
  const [currentAnimation, setCurrentAnimation] = useState(baseAnimation);

  useEffect(() => {
    const handleScroll = () => {
      const scrollPosition = window.scrollY;
      const windowHeight = window.innerHeight;

      if (scrollPosition > windowHeight * 0.5) {
        setCurrentAnimation('running');
      } else if (scrollPosition > windowHeight * 0.2) {
        setCurrentAnimation('walking');
      } else {
        setCurrentAnimation(baseAnimation);
      }
    };

    window.addEventListener('scroll', handleScroll);
    return () => window.removeEventListener('scroll', handleScroll);
  }, [baseAnimation]);

  return (
    <Player 
      skinUrl={skinUrl}
      animation={currentAnimation}
      width={200}
      height={280}
    />
  );
}
```

## Interactive Contact Form

```jsx
import { useState } from 'react';

function InteractiveContact() {
  const [formData, setFormData] = useState({
    name: '',
    email: '',
    message: ''
  });
  const [characterState, setCharacterState] = useState('idle');
  const [isSubmitting, setIsSubmitting] = useState(false);

  const handleChange = (e) => {
    const { name, value } = e.target;
    setFormData(prev => ({ ...prev, [name]: value }));
    
    // Change character animation based on form interaction
    if (value.length > 0 && characterState === 'idle') {
      setCharacterState('typing');
    }
  };

  const handleSubmit = async (e) => {
    e.preventDefault();
    setIsSubmitting(true);
    setCharacterState('sending');
    
    // Simulate form submission
    await new Promise(resolve => setTimeout(resolve, 2000));
    
    setCharacterState('success');
    setIsSubmitting(false);
    
    // Reset after success animation
    setTimeout(() => {
      setCharacterState('idle');
      setFormData({ name: '', email: '', message: '' });
    }, 3000);
  };

  const getCharacterAnimation = () => {
    switch (characterState) {
      case 'typing': return 'typing';
      case 'sending': return 'running';
      case 'success': return 'thumbsup';
      default: return 'idle';
    }
  };

  const getCharacterMessage = () => {
    switch (characterState) {
      case 'typing': return "I'm listening...";
      case 'sending': return "Sending your message!";
      case 'success': return "Message sent! I'll get back to you soon.";
      default: return "Let's work together!";
    }
  };

  return (
    <div className="interactive-contact">
      <div className="contact-character">
        <Player 
          skinUrl="https://example.com/contact-character.png"
          animation={getCharacterAnimation()}
          width={300}
          height={400}
        />
        <div className="speech-bubble">
          <p>{getCharacterMessage()}</p>
        </div>
      </div>
      
      <form onSubmit={handleSubmit} className="contact-form">
        <h2>Get In Touch</h2>
        
        <input
          type="text"
          name="name"
          placeholder="Your Name"
          value={formData.name}
          onChange={handleChange}
          required
        />
        
        <input
          type="email"
          name="email"
          placeholder="Your Email"
          value={formData.email}
          onChange={handleChange}
          required
        />
        
        <textarea
          name="message"
          placeholder="Your Message"
          value={formData.message}
          onChange={handleChange}
          rows={5}
          required
        />
        
        <button type="submit" disabled={isSubmitting}>
          {isSubmitting ? "Sending..." : "Send Message"}
        </button>
      </form>
    </div>
  );
}
```

## Performance Optimizations

### Lazy Loading Characters

```jsx
import { lazy, Suspense } from 'react';

const Player = lazy(() => import('adzyjs').then(module => ({ default: module.Player })));

function OptimizedCharacter({ skinUrl, ...props }) {
  return (
    <Suspense fallback={<div className="character-placeholder">Loading...</div>}>
      <Player skinUrl={skinUrl} {...props} />
    </Suspense>
  );
}
```

### Intersection Observer for Animations

```jsx
import { useEffect, useRef, useState } from 'react';

function ViewportAnimatedCharacter({ skinUrl, ...props }) {
  const [isVisible, setIsVisible] = useState(false);
  const characterRef = useRef();

  useEffect(() => {
    const observer = new IntersectionObserver(
      ([entry]) => setIsVisible(entry.isIntersecting),
      { threshold: 0.3 }
    );

    if (characterRef.current) {
      observer.observe(characterRef.current);
    }

    return () => observer.disconnect();
  }, []);

  return (
    <div ref={characterRef}>
      {isVisible && (
        <Player 
          skinUrl={skinUrl}
          animation="idle"
          {...props}
        />
      )}
    </div>
  );
}
```
