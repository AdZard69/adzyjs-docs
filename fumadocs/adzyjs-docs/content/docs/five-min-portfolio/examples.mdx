## Example 1: Minimalist Developer Portfolio

A clean, professional portfolio focusing on simplicity and performance.

### Preview Features
- Single-page layout
- Smooth scroll animations
- Interactive skill meters
- Contact form with character feedback

### Complete Code

```jsx
import { Player } from 'adzyjs';
import { useState, useEffect } from 'react';
import './MinimalistPortfolio.css';

function MinimalistPortfolio() {
  const [scrollY, setScrollY] = useState(0);

  useEffect(() => {
    const handleScroll = () => setScrollY(window.scrollY);
    window.addEventListener('scroll', handleScroll);
    return () => window.removeEventListener('scroll', handleScroll);
  }, []);

  return (
    <div className="minimalist-portfolio">
      {/* Navigation */}
      <nav className="nav">
        <div className="nav-brand">Alex Chen</div>
        <div className="nav-links">
          <a href="#about">About</a>
          <a href="#skills">Skills</a>
          <a href="#projects">Projects</a>
          <a href="#contact">Contact</a>
        </div>
      </nav>

      {/* Hero Section */}
      <section id="hero" className="hero">
        <div className="hero-content">
          <h1>Full Stack Developer</h1>
          <p>I create seamless digital experiences with modern web technologies</p>
          <button className="hero-cta">View My Work</button>
        </div>
        <div className="hero-character">
          <Player 
            skinUrl="https://example.com/professional-skin.png"
            animation={scrollY > 50 ? "thinking" : "idle"}
            width={280}
            height={400}
          />
        </div>
      </section>

      {/* About Section */}
      <section id="about" className="about">
        <div className="container">
          <div className="about-text">
            <h2>About Me</h2>

              </div>

                <h3>20+</h3>
                <p>Happy Clients</p>

            </div>
          </div>
          <div className="about-character">
            <Player 
              skinUrl="https://example.com/coding-skin.png"

              width={250}

      <div className="container">
        <h2>Technical Skills</h2>
        <div className="skills-grid">
          {skills.map((skill, index) => (
            <div key={index} className="skill-item">
              <div className="skill-character">
                <Player 
                  skinUrl={`https://example.com/${skill.name.toLowerCase()}-dev.png`}
                  animation="idle"
                  width={120}
                  height={180}
                />
              </div>
              <h3>{skill.name}</h3>
              <div className="skill-bar">
                <div 
                  className="skill-progress"
                  style={{ 
                    width: `${skill.level}%`,
                    backgroundColor: skill.color 
                  }}
                />
              </div>
              <span className="skill-percentage">{skill.level}%</span>
            </div>
          ))}
        </div>
      </div>
    </section>
  );
}

export default MinimalistPortfolio;
```

## Example 2: Gaming-Themed Portfolio

A vibrant portfolio with Minecraft-inspired design elements.

### Key Features
- Pixelated UI elements
- Block-style animations
- Character inventory system
- Achievement showcase

```jsx
import { Player } from 'adzyjs';
import { useState } from 'react';

function GamingPortfolio() {
  const [selectedCharacter, setSelectedCharacter] = useState(0);
  const [achievementsUnlocked, setAchievementsUnlocked] = useState([]);

  const characters = [
    { name: "Main Character", skin: "https://example.com/main.png", animation: "idle" },
    { name: "Builder", skin: "https://example.com/builder.png", animation: "building" },
    { name: "Explorer", skin: "https://example.com/explorer.png", animation: "walking" }
  ];

  const achievements = [
    { id: 1, name: "First Portfolio", description: "Created your first portfolio", icon: "üèÜ" },
    { id: 2, name: "Responsive Master", description: "Made portfolio mobile-friendly", icon: "üì±" },
    { id: 3, name: "Animation Expert", description: "Added character animations", icon: "üéÆ" }
  ];

  const unlockAchievement = (achievementId) => {
    if (!achievementsUnlocked.includes(achievementId)) {
      setAchievementsUnlocked([...achievementsUnlocked, achievementId]);
    }
  };

  return (
    <div className="gaming-portfolio">
      {/* Minecraft-style Header */}
      <header className="minecraft-header">
        <div className="health-bar">
          <span className="hearts">‚ù§Ô∏è‚ù§Ô∏è‚ù§Ô∏è‚ù§Ô∏è‚ù§Ô∏è</span>
        </div>
        <div className="character-selector">
          {characters.map((char, index) => (
            <div 
              key={index}
              className={`character-slot ${selectedCharacter === index ? 'selected' : ''}`}
              onClick={() => setSelectedCharacter(index)}
            >
              <Player 
                skinUrl={char.skin}
                animation="idle"
                width={50}
                height={70}
              />
            </div>
```

## Performance Optimization

### Lazy Loading Implementation

```javascript
import { lazy, Suspense } from 'react';

const LazyPlayer = lazy(() => import('adzyjs').then(module => ({ 
  default: module.Player 
})));

function OptimizedCharacter(props) {
  return (
    <Suspense fallback={<div className="character-loading">Loading...</div>}>
      <LazyPlayer {...props} />
    </Suspense>
  );
}
```

### Image Preloading

```javascript
const preloadSkins = async (skinUrls) => {
  const promises = skinUrls.map(url => {
    return new Promise((resolve, reject) => {
      const img = new Image();
      img.onload = resolve;
      img.onerror = reject;
      img.src = url;
    });
  });
  
  await Promise.all(promises);
};

// Use in component
useEffect(() => {
  const skins = [
    'https://example.com/skin1.png',
    'https://example.com/skin2.png'
  ];
  preloadSkins(skins);
}, []);
```
